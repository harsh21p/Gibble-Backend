 package com.practicum.authentication_service.service;
 
 import io.micrometer.core.instrument.Counter;
 import io.micrometer.core.instrument.MeterRegistry;
 import io.micrometer.core.instrument.Timer;
 import org.springframework.stereotype.Service;
 
 @Service
 public class JwtMetricsService {
     private final Counter tokenGenerationCounter;
     private final Counter expiredTokenCounter;
     private final Counter invalidTokenCounter;
     private final Timer tokenVerificationTimer;
     private final Counter userRegistrationCounter;
     private final Counter validTokenCounter;
 
     public JwtMetricsService(MeterRegistry registry) {
         this.tokenGenerationCounter = Counter.builder("jwt.token.generated")
                 .description("Total number of JWT tokens generated")
                 .register(registry);
 
         this.expiredTokenCounter = Counter.builder("jwt.token.expired")
                 .description("Total number of expired JWT tokens")
                 .register(registry);
 
         this.invalidTokenCounter = Counter.builder("jwt.token.invalid")
                 .description("Total number of invalid JWT tokens")
                 .register(registry);
 
         this.validTokenCounter = Counter.builder("jwt.token.validation")
                 .description("Total number of validation")
                 .register(registry);
 
         this.userRegistrationCounter = Counter.builder("user.registration")
                 .description("Total number of registered users")
                 .register(registry);
 
         this.tokenVerificationTimer = Timer.builder("jwt.token.verification.time")
                 .description("Time taken for JWT token verification")
                 .register(registry);
     }
 
     public void incrementTokenGenerationCount() {
         tokenGenerationCounter.increment();
     }
 
     public void incrementExpiredTokenCount() {
         expiredTokenCounter.increment();
     }
 
     public void incrementInvalidTokenCount() {
         invalidTokenCounter.increment();
     }
 
     public Timer getTokenVerificationTimer() {
         return tokenVerificationTimer;
     }
 
     public void incrementUserRegistrationCounter() {userRegistrationCounter.increment();}
 
     public void incrementValidTokenCounter() {validTokenCounter.increment();}
 }
